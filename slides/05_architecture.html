func (e *Engine) Evaluate(
  ctx, agent, toolName, request
) (Decision, error) {
  // 1. Check cache first
  if decision, ok := e.cache.Get(key); ok {
    return e.applyMode(decision), nil
  }
  // 2. Look up policy
  policy := e.policies[agent.AgentType]
  // 3. Evaluate
  decision := e.evaluatePolicy(policy, toolName)
  // 4. Cache + Audit
  e.cache.Set(key, decision)
  e.emitAudit(agent, tool, decision)
  return e.applyMode(decision), nil
}

Critical Design Decision

The policy engine is IN the router, not a sidecar. If it were a sidecar, there would be a bypass path. By embedding it in the router, we ensure complete mediation - every tool call goes through policy evaluation.
